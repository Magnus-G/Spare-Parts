SCSS Styles for Icon fonts

Light from above

<div class="style-light-from-above"> <i class="icon icon-star"></i> </div> -------------------------------------------------------------------- @mixin text-gradient ($top, $bottom) { color: $bottom; background: -webkit-linear-gradient($top, $bottom); -webkit-background-clip: text; -webkit-text-fill-color: transparent; } @mixin blend-layer($color, $edge-sharpness, $transparentize) { // covers the whole icon and adds drop shadow. text-shadow: 0 1px $edge-sharpness transparentize($color, $transparentize); } .style-light-from-above { // remove :before if using style for text and not icon font .icon-star:before { // text-shadow with transparency @include blend-layer(red, 2px, 0.5); // this has to be on the actual font content @include text-gradient(white, darken(red, 10)); } }
